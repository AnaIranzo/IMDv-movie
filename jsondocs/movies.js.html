const mongoose = require('mongoose');
    /**
  * Descripción de la función: Devuelve el nombre del esquema de la base de datos para los objetos de ámbito de esquema.
  * @memberof SQLSchema
  * @method objectSchema
  * @async
  * @throws {error} Error en la creacion de la pelicula en la BBDD
  */

const objectSchema ={

    id: {
        type: String, 
        required: true,
        unique: true               
    },
    fullTitle: {
        type: String, 
        required: true,
    },
    year: {
        type: String, 
        required: true,
    },
    image:{
        type: String,
        validate: {
            validator: function(url){
                if(url.indexOf('.jpg') != -1 || url.indexOf('.png') != -1)
                    return true;
                else {
                    return false;
                }
            }, 
            message: "Sólo imágenes JPG o PNG"
        }
    },
    runtimeStr: {
        type: String, 
        required: true,
    },
    plot: {
        type: String, 
        required: true,
    },
    directors: {
        type: String, 
        required: true,
    },
    actorList: {
        type: String,
    },
    genres: {
        type: String, 
        
    },
    imDbRating: {
        type: String, 
        
    },

    opinions: {
        type: String, 
        
    },//"LAS OPINIONES LAS OBTENEMOS CON SCRAPPING"
}


// Crear el esquema
const movieSchema = mongoose.Schema(objectSchema);
// Crear el modelo --> Colección
const Movie = mongoose.model('Movie', movieSchema);

module.exports = Movie;
